  <script>
    function updateLinks(bankId) {
      if (!bankId) {
        alert("Veuillez sélectionner une banque");
        return;
      }
      $('#card-cheques').attr('href', `/tresorerie/cheques/banque/${bankId}`);
      $('#card-effets').attr('href', `/tresorerie/effets/banque/${bankId}`);
      $('#card-releves').attr('href', `/tresorerie/releves/banque/${bankId}`);
    }

    $(document).ready(function () {
      $('#bank').on('change', function () {
        const selectedBankId = $(this).val();
        updateLinks(selectedBankId);
      });

      $('.achat-card').on('click', function () {
        const $icon = $(this).find('.card-icon-wrapper i');
        $icon.addClass('fa-spin');
        setTimeout(() => $icon.removeClass('fa-spin'), 800);
      });

      $('#formAjouterBanque').on('submit', function (e) {
        e.preventDefault();
        const banqueData = {
          name: $('#bankName').val(),
          rib: $('#bankRib').val(),
          agence: $('#bankAgence').val()
        };
        $.ajax({
          type: 'POST',
          url: '/tresorerie/banques/create',
          data: banqueData,
          success: function (response) {
            alert('Banque ajoutée avec succès');
            $('#ajouterBanqueModal').modal('hide');
            location.reload();
          },
          error: function (err) {
            console.error(err);
            alert("Erreur lors de l'ajout de la banque");
          }
        });
      });
    });
  </script>
