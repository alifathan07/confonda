<script>
  // Enhanced bank selection functionality
  $(document).ready(function() {
    $('#bank').on('change', function() {
      const $this = $(this);
      if ($this.val()) {
        $this.css({
          'border-color': '#4e73df',
          'background-color': '#f8f9fc'
        });
      } else {
        $this.css({
          'border-color': '#e3e6f0',
          'background-color': 'white'
        });
      }
    });

    // Add loading effect to action cards
    $('.achat-card').on('click', function(e) {
      const $icon = $(this).find('.card-icon-wrapper i');
      $icon.addClass('fa-spin');

      setTimeout(function() {
        $icon.removeClass('fa-spin');
      }, 800);
    });
  });
  $('#formAjouterBanque').on('submit', function (e) {
  e.preventDefault();

  const banqueData = {
    name: $('#bankName').val(),
    rib: $('#bankRib').val(),
    agence: $('#bankAgence').val()
  };

  $.ajax({
    type: 'POST',
    url: '/tresorerie/banques/create', // Make sure this endpoint exists in your backend
    data: banqueData,
    success: function (response) {
      alert('Banque ajoutée avec succès');
      $('#ajouterBanqueModal').modal('hide');
      location.reload(); // Optional: reload to refresh the bank select
    },
    error: function (err) {
      console.error(err);
      alert("Erreur lors de l'ajout de la banque");
    }
  });
});

</script>
